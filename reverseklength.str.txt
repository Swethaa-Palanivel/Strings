import java.util.*;
public class Main{
    public static void main(String[]args){
        Scanner sc = new Scanner(System.in);
        String S =sc.nextLine();
        int k =sc.nextInt();
        String ans = "";
        for(int i =0;i<S.length();i+=k){
            int end = Math.min(i+k,S.length());
            for(int j = end - 1;j>=i;j--){
                ans += S.charAt(j);
            }
        }
        System.out.print(ans);
    }
}
